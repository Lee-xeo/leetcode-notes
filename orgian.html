def maxSubArray(nums):
    dp = [0] * len(nums)
    dp[0] = nums[0]
    maxn = dp[0]
    for i in range(1, len(nums)):
        dp[i] = max(nums[i], dp[i-1] + nums[i])
        maxn = max(maxn, dp[i])
    return maxn
